00026 med_int 
00027 MEDnValProfil(med_idt fid, char *nom)
00028 {
00029   med_int n = 0;
00030   med_idt gid;
00031   char chemin[MED_TAILLE_PROFILS+MED_TAILLE_NOM+1]; 
00032   med_err ret;
00033   
00034   /*
00035    * On inhibe le gestionnaire d'erreur HDF 5
00036    */
00037   _MEDmodeErreurVerrouiller();
00038 if (MEDcheckVersion(fid) < 0) return -1;
00039 
00040 
00041   /* 
00042    * ouverture du groupe /PROFILS/<nom>
00043    */  
00044   strcpy(chemin,MED_PROFILS);
00045   strcat(chemin,nom); 
00046   if ((gid = _MEDdatagroupOuvrir(fid,chemin)) < 0){
00047     MESSAGE("Impossible d'ouvrir le groupe MED_PROFILS : ");
00048     SSCRUTE(chemin); return -1;
00049   }
00050   
00051   if ((ret = _MEDattrEntierLire(gid,MED_NOM_NBR,&n)) < 0) {
00052     MESSAGE("Erreur à la lecture de l'attribut MED_NOM_NBR : ");
00053     ISCRUTE(n); return -1;
00054   }
00055   
00056   /*
00057    * On ferme tout
00058    */
00059   if ((ret = _MEDdatagroupFermer(gid)) < 0)
00060     return -1; 
00061 
00062   return n;
00063 }
